## I. Bake Me A Wish Coding Exercise

### CakeOrder.java
```java
import java.util.HashMap; 
import java.util.Map;

public class CakeOrder {
    private Map<String, Double> orderMap = new HashMap<>();

    public Map<String, Double> getOrderMap() { 
        return orderMap; 
    }

    public void setOrderMap(Map<String, Double> orderMap) { 
        this.orderMap = orderMap; 
    }

    public void addOrderDetails(String orderId, double cakeCost) { 
        orderMap.put(orderId, cakeCost); 
    }

    public Map<String, Double> findOrdersAboveSpecifiedCost(double cakeCost) { 
        Map<String, Double> filterOrders = new HashMap<>();
        for (Map.Entry<String, Double> entry : orderMap.entrySet()) {
            if (entry.getValue() > cakeCost) { 
                filterOrders.put(entry.getKey(), entry.getValue());
            }
        }
        return filterOrders;
    }
}
```

### UserInterface.java
```java
import java.util.HashMap; 
import java.util.Map; 
import java.util.Scanner;

public class UserInterface {
    public static void main(String[] args) { 
        Scanner sc = new Scanner(System.in);
        CakeOrder cakeOrder = new CakeOrder(); 
        System.out.println("Enter number of cake orders to be added"); 
        int input = sc.nextInt();
 
        System.out.println("Enter the cake order details (Order Id: CakeCost)"); 
        for (int i = 0; i < input; i++) {
            String orderDetails = sc.next(); 
            String[] parts = orderDetails.split(":"); 
            String orderId = parts[0];
            double cakeCost = Double.parseDouble(parts[1]);

            cakeOrder.addOrderDetails(orderId, cakeCost);
        }

        System.out.println("Enter the cost to search the cake orders"); 
        double search = sc.nextDouble();
        Map<String, Double> results = cakeOrder.findOrdersAboveSpecifiedCost(search); 
        if (results.isEmpty()) {
            System.out.println("No cake orders found");
        } else {
            System.out.println("Cake Orders above the specified cost"); 
            for (Map.Entry<String, Double> entry : results.entrySet()) {
                System.out.println("Order ID: " + entry.getKey() + ", Cake Cost: " + entry.getValue());
            }
        }
    }
}
```

## II. Forensic Report Coding Exercise

### ForensicReport.java
```java
import java.util.ArrayList; 
import java.util.Date; 
import java.util.HashMap; 
import java.util.List; 
import java.util.Map; 

public class ForensicReport {
    private Map<String, Date> reportMap = new HashMap<>();

    public Map<String, Date> getReportMap() {
        return reportMap;
    }

    public void setReportMap(Map<String, Date> reportMap) { 
        this.reportMap = reportMap; 
    }

    public void addReportDetails(String reportingOfficer, Date reportFiledDate) {
        reportMap.put(reportingOfficer, reportFiledDate);
    }

    public List<String> getOfficersWhoFiledReportsOnDate(Date reportFiledDate) {
        List<String> names = new ArrayList<>();
        for (Map.Entry<String, Date> entry : reportMap.entrySet()) {
            if (reportFiledDate.equals(entry.getValue())) {
                names.add(entry.getKey());
            }
        }
        return names;
    }
}
```

### UserInterface.java
```java
import java.text.ParseException; 
import java.text.SimpleDateFormat; 
import java.util.Date; 
import java.util.List; 
import java.util.Scanner;

public class UserInterface {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        SimpleDateFormat df = new SimpleDateFormat("yyyy-mm-dd"); 
        ForensicReport f = new ForensicReport(); 
        System.out.println("Enter number of reports to be added");
        int num = sc.nextInt();
        System.out.println("Enter the Forensic reports (Reporting Officer: Report Filed Date)"); 
        try {
            for (int i = 0; i < num; i++) {
                String input = sc.next();
                String details[] = input.split(":");
                String officer = details[0]; 
                String fileDate = details[1]; 
                Date d = df.parse(fileDate); 
                f.addReportDetails(officer, d);
            }
            System.out.println("Enter the filed date to identify the reporting officers"); 
            String s = sc.next();
            Date iDate = df.parse(s);
            List<String> l = f.getOfficersWhoFiledReportsOnDate(iDate); 
            if (l.isEmpty()) {
                System.out.println("No reporting officer filed the report");
            } else {
                System.out.println("Reports filed on the " + s + " are by"); 
                for (String names : l) {
                    System.out.println(names);
                }
            }
        } catch (ParseException e) {
            e.printStackTrace();
        }
    }
}
```

## III. Heaven Homes Coding Exercise

### Apartment.java
```java
import java.util.HashMap; 
import java.util.Map;

public class Apartment {
    private Map<String, Double> apartmentDetailsMap;

    public Apartment() {
        apartmentDetailsMap = new HashMap<>();
    }

    public Map<String, Double> getApartmentDetailsMap() { 
        return apartmentDetailsMap; 
    }

    public void setApartmentDetailsMap(Map<String, Double> apartmentDetailsMap) { 
        this.apartmentDetailsMap = apartmentDetailsMap; 
    }

    public void addApartmentDetails(String apartmentNumber, double rent) { 
        apartmentDetailsMap.put(apartmentNumber, rent);
    }

    public double findTotalRentOfApartmentsInTheGivenRange(double minimumRent, double maximumRent) { 
        double sum = 0;
        for (Map.Entry<String, Double> entry : apartmentDetailsMap.entrySet()) {
            if (entry.getValue() >= minimumRent && entry.getValue() <= maximumRent) { 
                sum += entry.getValue();
            }
        }
        return sum;
    }
}
```

### UserInterface.java
```java
import java.util.Scanner;

public class UserInterface {
    public static void main(String[] args) { 
        Scanner sc = new Scanner(System.in);
        System.out.println("Enter number of details to be added "); 
        int nDetails = sc.nextInt();
        sc.nextLine(); // Consume the newline character 
        Apartment apartment = new Apartment();
        System.out.println("Enter the details (Apartment number: Rent) "); 
        for (int i = 0; i < nDetails; i++) {
            String allDetails = sc.nextLine(); // Use nextLine() to read the entire line 
            String[] parts = allDetails.split(":");
            String apartNumber = parts[0];
            double rent = Double.parseDouble(parts[1]);
            apartment.addApartmentDetails(apartNumber, rent);
        }

        System.out.println("Enter the range to filter the details "); 
        double minRange = sc.nextDouble();
        double maxRange = sc.nextDouble();

        double result = apartment.findTotalRentOfApartmentsInTheGivenRange(minRange, maxRange);
        if (result > 0) {
            System.out.printf("Total Rent in the range %.1f to %.1f USD: %.1f%n", minRange, maxRange, result);
        } else {
            System.out.println("No apartments found in this range");
        }
    }
}
```
