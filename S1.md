```java
import java.util.ArrayList;
import java.util.Date;
import java.util.HashMap;
import java.util.List;
import java.util.Map;
import java.util.Scanner;

public class BookCabMain {

    private Map<String, Date> bookingMap = new HashMap<>();

    public Map<String, Date> getBookingMap() {
        return bookingMap;
    }

    public void setBookingMap(Map<String, Date> bookingMap) {
        this.bookingMap = bookingMap;
    }

    public List<String> findCabBookingIdsForTheBookingDate(Date date) {
        List<String> bookingIds = new ArrayList<>();
        for (Map.Entry<String, Date> entry : bookingMap.entrySet()) {
            if (entry.getValue().equals(date)) {
                bookingIds.add(entry.getKey());
            }
        }
        return bookingIds;
    }
}
```

```java
import java.text.ParseException;
import java.text.SimpleDateFormat;
import java.util.Date;
import java.util.HashMap;
import java.util.List;
import java.util.Map;
import java.util.Scanner;

public class UserInterface {
    public static void main(String[] args) throws ParseException {
        BookCabMain cab = new BookCabMain();
        Map<String, Date> map = new HashMap<>();
        Scanner sc = new Scanner(System.in);
        
        System.out.println("Enter number of bookings to be added:");
        int n = sc.nextInt();
        
        System.out.println("Enter the Booking details (Booking Id:Date of Booking in yyyy-MM-dd):");
        String[] bookingDetails = new String[n];
        for (int i = 0; i < n; i++) {
            bookingDetails[i] = sc.next();
        }

        SimpleDateFormat sdf = new SimpleDateFormat("yyyy-MM-dd");

        for (String bookingDetail : bookingDetails) {
            String[] parts = bookingDetail.split(":");
            map.put(parts[0], sdf.parse(parts[1]));
            cab.setBookingMap(map);
        }

        System.out.println("Enter the date to check the booking (yyyy-MM-dd):");
        String dateString = sc.next();
        Date date = sdf.parse(dateString);
        
        List<String> bookingIds = cab.findCabBookingIdsForTheBookingDate(date);
        if (bookingIds.isEmpty()) {
            System.out.println("No bookings were made on " + dateString);
        } else {
            System.out.println("Booking Ids booked on " + dateString + " are:");
            for (String bookingId : bookingIds) {
                System.out.println(bookingId);
            }
        }
    }
}
```

### Solution 2: Strings

**UserInterface.java**

```java
import java.util.Scanner;

public class UserInterface {

    public static boolean checkCountNo(String s) {
        return s.matches("[0-9]{2}");
    }

    public static boolean checkSeatNo(String s) {
        return s.matches("[A-z]{1}[0-9]{2}");
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        String train = sc.nextLine().trim();

        if (train.length() != 10) {
            System.out.println("Invalid Input");
            return;
        }

        String s = train.substring(5, 7);
        String m = train.substring(7, 10);

        boolean isValid = false;

        if (checkCountNo(s)) {
            if (checkSeatNo(m)) {
                isValid = true;
            } else {
                System.out.println("Invalid Seat Number");
                return;
            }
        } else {
            System.out.println("Invalid Count");
            return;
        }

        int count = Integer.parseInt(s);
        if (count > 10) {
            System.out.println("Invalid Count");
            return;
        }

        if (isValid) {
            int seatNumber = Integer.parseInt(train.substring(8, 10)) - 1;
            for (int i = 0; i < count; i++) {
                System.out.print(train.charAt(7));
                System.out.println(++seatNumber);
            }
        }
    }
}
```

### Solution 4: Streams

**Dish.java**
```java
public class Dish {
    private String dishName;
    private String cuisine;
    private double price;
    private double rating;

    public Dish(String dishName, String cuisine, double price, double rating) {
        this.dishName = dishName;
        this.cuisine = cuisine;
        this.price = price;
        this.rating = rating;
    }

    public String getDishName() {
        return dishName;
    }

    public String getCuisine() {
        return cuisine;
    }

    public double getPrice() {
        return price;
    }

    public double getRating() {
        return rating;
    }

    @Override
    public String toString() {
        return dishName + "|" + cuisine + "|" + price + "|" + rating;
    }
}
```

**DishUtil.java**
```java
import java.util.List;
import java.util.stream.Collectors;
import java.util.stream.Stream;

public class DishUtil {
    public List<Dish> getDishesByCuisine(Stream<Dish> dishStream, String cuisine) {
        return dishStream.filter(dish -> dish.getCuisine().equalsIgnoreCase(cuisine))
                         .collect(Collectors.toList());
    }
}
```

**UserInterface.java**
```java
import java.util.ArrayList;
import java.util.List;
import java.util.Scanner;

public class UserInterface {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        List<Dish> dishList = new ArrayList<>();
        DishUtil dishUtil = new DishUtil();

        System.out.println("Enter the number of dishes:");
        int numDishes = sc.nextInt();

        System.out.println("Enter the dish details (Name:Cuisine:Price:Rating):");
        for (int i = 0; i < numDishes; i++) {
            sc.nextLine();
            String[] details = sc.nextLine().split(":");
            String dishName = details[0];
            String cuisine = details[1];
            double price = Double.parseDouble(details[2]);
            double rating = Double.parseDouble(details[3]);

            dishList.add(new Dish(dishName, cuisine, price, rating));
        }

        System.out.println("Enter the cuisine to filter by:");
        String cuisineFilter = sc.next();

        List<Dish> filteredDishes = dishUtil.getDishesByCuisine(dishList.stream(), cuisineFilter);
        
        if (!filteredDishes.isEmpty()) {
            System.out.println("Dishes of cuisine " + cuisineFilter + " are:");
            filteredDishes.forEach(System.out::println);
        } else {
            System.out.println("No dishes found for the given cuisine " + cuisineFilter);
        }
    }
}
```
